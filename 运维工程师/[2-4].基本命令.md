
# 基本命令


```shell 
        1. ip address show / ip a          检查网卡地址配置
        2. ping                            网络测试
        
        3. shutdown                        系统关机
            -h  指定关机时间        shutdown -h 60
            -c  取消    
            -r  重启               shutdown -r 5
        4. halt                            直接关机
        5. poweroff                        直接关机
        6. reboot                          重启

        7. ls  /etc/                       检查文件或目录是否存在
            -d 有针对的查看目录信息  ls -d /etc
            ll -d /opt 
            ll -ia /opt                    查看目录下的文件信息
        8. mkdir                           创建目录
            -p                          递归创建目录
        
        9. cd                              切换目录
            cd .. 退回到上一级
            cd ../../
            cd - 返回到上次的目录
            cd ~ 返回到/home目录

        10. pwd                             显示当前目录

        11. man

        12. touch                           创建出一个空文件
            touch /opt/file
        13. vi                              编辑文件
            vi /opt/file 
            第一个步骤：进入编辑模式（插入模式）
            按键盘小写字母 i 
            第二个步骤：编写文件内容
            第三个步骤：进行保存关闭文档
            按esc（退出编辑状态）输入:wq    (保存/关闭)
                                   :q     (不保存直接退出)
                                   :w
            快捷操作：
                dd      (剪切行)
                3dd     (剪切三行)
                u       (还原)
                p       (粘贴)
                3p      (粘贴3行)
                yy      (复制一行内容)
                3yy     (复制三行内容)
            
            vi三种模式:
                命令模式（编辑模式）
                插入模式
                底行模式

                命令模式 ----> 插入模式
                    i   -- 表示从光标所在位置进入编辑模式
                    I   -- 表示将光标移动到一行的行首，在进入编辑状态
                    o   -- 在光标的行下，新建一行编辑
                    O   -- 在光标的行上，新建一行编辑
                    a   -- 将光标移动到右边的下一个字符，进行编辑
                    A   -- 将光标移动到行尾，进入编辑状态
                    C   -- 将光标到行尾内容进行删除，并进入编辑状态
                    cc  -- 将这整行内容删除，并进入编辑模式 

                    只移动光标，不进入编辑状态
                        G   -- 将光标快速移动切换尾部
                        gg  -- 将光标快速移动切换首部
                        ngg -- n表示移动到第几行
                        $   -- 将光标移动到一行的结尾
                        0/^ -- 将光标移动到一行的行首

                插入模式 ----> 命令模式
                    esc
                
                命令模式 ----> 底行模式
                    :         -- 输入命令
                    :set nu   -- 显示行号
                    /         -- 搜索（向下 n一次向下）
                    ?         -- 搜索（向上 n一次向上）


        14. echo                            信息打印到屏幕上
            echo "hello world" > /opt/file2  将信息直接存放到文件里(替换文件内容)
            echo "hello world" >> /opt/file2 将信息直接存放到文件里(追加文件内容)
        
        15. cat                             查看文件内容

        16. cp                              复制数据
            cp /opt/file /opt/file_copy               复制文件
            cp -r /opt/directory /opt/directory_copy  复制目录
            \cp -r /opt/directory /opt/directory_copy 复制目录覆盖存在文件

        17. rm                              删除文件
            rm file.txt 
            \rm file.txt  强制直接删除
            rm -f file.txt 强制直接删除
            rm -rf /opt/directory_copy  删除文件

        18. mount                           挂载光驱
            mount /dev/cdrom /mnt 
        19. umount                          卸载挂载点
            umount /mnt 

        20. yum                             安装软件
            a. 指定yum源文件
                https://opsx.alibaba.com/mirror 
                /yum.repos.d/CentOS-Base.repo 
            b. yum install [名称]
                yum install -y vim tree wget net-tools nmap bash-completion

        21. rpm                             安装软件包
                rpm -qa              查询软件包是否安装
                rpm -qf file         查询文件在哪个文件包


        22. ps -ef                          查看进程信息
        23. kill [pid]                      删除进程
                kill -9 [pid]               强制删除进程


        24. tail                            查看文件后几行
            tail -6 /etc/profile 
            tail -f /etc/profile     追踪查看文件内容变换
        25. head                            查看文件前几行
            head -6 /etc/profile   


        26. free                            内存信息
                free -h 
        27. lscpu                           cpu信息
        28. df                              磁盘信息
                df -h
        29. w                               负载信息


        30. useradd                         创建用户
            useradd 用户名
        31. passwd                          设置用户密码命令
            passwd 用户名
        32. su                              切换用户
            su - 用户名                 
        33. id                              检查创建的用户是否存在
            id 用户名
        34. whoami                          确认用户身份


        35. 防火墙
            centos6:
                /etc/init.d/iptables status/stop/start  -- 临时关闭
                chkconfig iptables off                  -- 永久关闭 
            centos7:
                systemctl status/stop/start/is-active firewalld     -- 临时关闭
                systemctl disable firewalld                         -- 永久关闭

        36. selinux服务程序
            centos6 == centos7
                getenforce                              -- selinnux状态
                setenforce  1/0                         -- 1 临时开启，0 临时关闭
                vi /etc/selinux/chkconfig               -- 永久关闭, 只能重启启动
        
        37. source                          立即加载文件配置信息
                                            /etc/profile
                                            /etc/bashrx
                                            ~/.bashrc
                                            ~/.bashrc_profile
    
        38. file /etc/filename              区分文件类型
        39. ln                              创建文件链接
            ln -s /opt/file /usr/local/file_link    

        40. which                           查找文件的绝对路径
            which cat
        41. locate 文件名称                  显示文件路径
        42. find                            查找文件
            find / -name filename 
            find / -name filename -type f -size +1M
                                                  b byte 
                                                  c bytes(推荐)
                                                  w words
                                                  k kilobytes
                                                  M Megabytes
                                                  G Gigabytes
                -maxdepth 1 -mindepth       目录层级查找文件
                -perm                       文件权限查找数据
                    -perm 644
                -inum                       查找文件的硬链接
                    -inum 33584339
            find / -type f -name "*.txt" -delete    统一删除                    
                                         --exec rm -rf {} \;
                                         --exec cp {} /tmp \;
                                         | xargs rm -f
                                         | xargs -i cp {} /opt 
                                         | xargs cp -t /opt 
                                         | xargs -i nv {} /opt 
                                         | xargs nv  -t /opt 


        43. tree                            显示目录结构信息
            tree /opt
            tree -L 2 /         -- 查看目录结构层次信息
            tree -d /opt        -- 显示目录中左右结构目录

        44. tar                             压缩命令
                -z              -- 压缩方式 zip
                -c              -- 创建压缩包文件
                -v              -- 显示压缩过程
                -f              -- 指定压缩包文件路径信息
            tar -zcvf opt.tar.gz /opt       压缩
            tar -xcf opt.tar.gz             解压
            tar -tf opt.tar.gz              不解压的情况下查看压缩包的内容
            tar -zcvf opt.tar.gz ./opt --exclude=/opt/file1 
            tar -zcvf opt.tar.gz ./opt --exclude-from=/opt/file_exclude.txt 

        45. diff                            比较文件内容差异
            diff /opt/file1 /opt/file2 

        46. date                            时间命令
            date +%F 
            date "+%F %T"
            date "+%y-%m-%d %H:%M:%S"
            cp file.log file.log-$(date "+%F_%T")

        47. grep                            文件中查找信息
            -o                                  -- 显示命令执行过程
            -E                                  -- 允许正则的一些符号使用
            grep hello /opt/file                -- 查找hello这个单词是否存在于文件
            grep -v hello /opt/file             -- 排除hello，打印出文件信息
            grep -v "^#" /etc/selinux/conifg    -- 排除以#开头的信息
            grep "0*" /opt/file                 -- 查找*前的符号连续出现
```
